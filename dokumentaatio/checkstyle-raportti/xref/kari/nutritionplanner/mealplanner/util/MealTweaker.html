17  package kari.nutritionplanner.mealplanner.util;
18  
19  import kari.nutritionplanner.mealplanner.domain.Meal;
20  
21  /**
22   * CalculateMealin apuluokka, joka pyörittelee luodun aterian raaka-aineita sekä
23   * lopuksi pyöristää arvot järkevään muotoon.
24   *
25   * @author kari
26   */
27  public class MealTweaker {
28  
29      private final Meal meal;
30      private final double calories;
31      private final double protein;
32      private final double fat;
33      private final MacroCalculator mc;
34  
35      /**
36       * Konstuktori saa Meal-olion jota pitäisi säätää, sekä tavoitemakrot.
37       *
38       * @param meal Meal-olio, jolle säädöt pitäisi tehdä
39       * @param calories haluttu määrä kaloreja
40       * @param protein haluttu määrä proteiinin
41       * @param fat haluttu määrä rasvaa
42       */
43      public MealTweaker(Meal meal, double calories, double protein, double fat) {
44          this.meal = meal;
45          this.calories = calories;
46          this.protein = protein;
47          this.fat = fat;
48          this.mc = new MacroCalculator();
49      }
50  
51      /**
52       * Tekee kaikki taiat. Pyörittelee makroja ja säätää niitä, lopuksi
53       * pyöristää määrät järkevään muotoon.
54       *
55       */
56      public void tweakMeal() {
57          evenUpIngredients();
58          roundUpIngredients();
59      }
60  
61      private void proteinAlterer() {
62          if (!proteinOk()) {
63              double proteinToSub = protein - meal.getProtein();
64              if (meal.getSideIngredient().getProtein() > 3 && protein < meal.getProtein()) {
65                  subFromBoth(proteinToSub);
66              } else {
67                  subFromMain(proteinToSub);
68              }
69          }
70      }
71  
72      private void subFromMain(double proteinToSub) {
73          double mainAmount = mc.calculateAmountForProtein(proteinToSub, meal.getMainIngredient());
74          if (meal.getMainIngredientAmount() + mainAmount > 0.5) {
75              meal.setMainIngredientAmount(meal.getMainIngredientAmount() + mainAmount);
76          } else if (meal.getMainIngredient().getProtein() > 23) {
77              meal.setMainIngredientAmount(0.5);
78          } else {
79              meal.setMainIngredientAmount(0.8);
80          }
81      }
82  
83      private void subFromBoth(double proteinToSub) {
84          double mainAmount = mc.calculateAmountForProtein(proteinToSub / 4 * 3, meal.getMainIngredient());
85          double sideAmount = mc.calculateAmountForProtein(proteinToSub / 4, meal.getSideIngredient());
86          if (meal.getMainIngredientAmount() + mainAmount > 0.5 && meal.getSideIngredientAmount() + sideAmount > 0) {
87              meal.setMainIngredientAmount(meal.getMainIngredientAmount() + mainAmount);
88              meal.setSideIngredientAmount(meal.getSideIngredientAmount() + sideAmount);
89          } else {
90              subFromMain(proteinToSub);
91          }
92          if (meal.getMainIngredientAmount() < 0.5) {
93              meal.setMainIngredientAmount(0.5);
94          }
95      }
96  
97      private void caloriesTweaker() {
98          if (calories < meal.getCalories()) {
99              subCalories();
100         } else {
101             addCalories();
102         }
103     }
104 
105     private void fatSubber() {
106         if (!fatOk()) {
107             double fatToSub = fat - meal.getFat();
108             meal.setSauceAmount(meal.getSauceAmount() + mc.calculateAmountForFat(fatToSub, meal.getSauce()));
109             if (meal.getSauceAmount() < 0.1) {
110                 meal.setSauceAmount(0);
111             }
112         }
113     }
114 
115     private void evenUpIngredients() {
116         for (int i = 0; i < 50; i++) {
117             if (allOk()) {
118                 break;
119             }
120             fatSubber();
121             proteinAlterer();
122             caloriesTweaker();
123         }
124     }
125 
126     private boolean allOk() {
127         return caloriesOk() && proteinOk() && fatOk();
128     }
129 
130     /**
131      * Pyöristää aterian raaka-aineiden määrät kahden desimaalin tarkkuuteen.
132      * Esim. 1.23 on 123 grammaa.
133      */
134     public void roundUpIngredients() {
135         double main = (Math.ceil(meal.getMainIngredientAmount() * 100)) / 100;
136         double side = Math.ceil(meal.getSideIngredientAmount() * 100) / 100;
137         double sauce = Math.ceil(meal.getSauceAmount() * 100) / 100;
138         meal.setMainIngredientAmount(main);
139         meal.setSideIngredientAmount(side);
140         meal.setSauceAmount(sauce);
141     }
142 
143     private void subCalories() {
144         double toSub = calories - meal.getCalories();
145         // jos protskut ja rasva ok, vähennä vain lisäkettä, jos sitä on
146         if (proteinOk() && fatOk() && meal.getSideIngredientAmount() > 0.05) {
147             alterCaloriesFromSide(toSub);
148             // jos protskut ok, vähennetään lisäkettä ja soosia, jos lisäkettä on
149         } else if (proteinOk() && meal.getSideIngredientAmount() > 0.05) {
150             alterCaloriesFromSideAndSauce(toSub);
151             // jos protskut tai rasva ok ja joko soosi tai lisäke on lähellä nollaa, vähennetään pääraaka-ainetta
152         } else if (proteinOk() && (meal.getSideIngredientAmount() < 0.1 || meal.getSauceAmount() < 0.1)) {
153             alterCaloriesFromMain(toSub);
154             // jos rasva ok, vähennetään pääraaka-ainetta ja lisuketta, jos lisuketta on
155         } else if (fatOk() && meal.getSideIngredientAmount() > 0.05) {
156             alterCaloriesFromMainAndSide(toSub);
157             // tai sitten vähennetään vähän kaikkia
158         } else {
159             alterCaloriesFromAll(toSub);
160         }
161         if (meal.getSideIngredientAmount() < 0) {
162             meal.setSideIngredientAmount(0);
163         }
164         if (meal.getSauceAmount() < 0) {
165             meal.setSauceAmount(0);
166         }
167     }
168 
169     private void addCalories() {
170         double toAdd = calories - meal.getCalories();
171         // jos protskut ja rasva ok, lisää vain lisäkettä
172         if (proteinOk() && fatOk()) {
173             alterCaloriesFromSide(toAdd);
174         } else if (proteinOk()) {
175             alterCaloriesFromSideAndSauce(toAdd);
176         } else {
177             alterCaloriesFromAll(toAdd);
178         }
179     }
180 
181     private boolean proteinOk() {
182         return (meal.getProtein() <= protein + 2 && meal.getProtein() >= protein - 2);
183     }
184 
185     private boolean caloriesOk() {
186         return (meal.getCalories() <= calories + 20 && meal.getCalories() >= calories - 20);
187     }
188 
189     private boolean fatOk() {
190         return (meal.getFat() <= fat + 2 && meal.getFat() >= fat - 2);
191     }
192 
193     private void alterCaloriesFromSide(double toAlter) {
194         double amountToSubFromSide = mc.calculateAmountForCalories(toAlter, meal.getSideIngredient());
195         meal.setSideIngredientAmount(meal.getSideIngredientAmount() + amountToSubFromSide);
196     }
197 
198     private void alterCaloriesFromSideAndSauce(double toAlter) {
199         double amountToSubFromSide = mc.calculateAmountForCalories((toAlter / 8 * 5), meal.getSideIngredient());
200         double amountToSubFromSauce = mc.calculateAmountForCalories((toAlter / 8 * 3), meal.getSauce());
201         meal.setSideIngredientAmount(meal.getSideIngredientAmount() + amountToSubFromSide);
202         meal.setSauceAmount(meal.getSauceAmount() + amountToSubFromSauce);
203     }
204 
205     private void alterCaloriesFromAll(double toAlter) {
206         double amountToSubFromSide = mc.calculateAmountForCalories((toAlter / 8 * 5), meal.getSideIngredient());
207         double amountToSubFromSauce = mc.calculateAmountForCalories((toAlter / 8 * 2), meal.getSauce());
208         double amountToSubFromMain = mc.calculateAmountForCalories(toAlter / 8, meal.getMainIngredient());
209         meal.setSideIngredientAmount(meal.getSideIngredientAmount() + amountToSubFromSide);
210         meal.setSauceAmount(meal.getSauceAmount() + amountToSubFromSauce);
211         meal.setMainIngredientAmount(meal.getMainIngredientAmount() + amountToSubFromMain);
212     }
213 
214     private void alterCaloriesFromMainAndSide(double toAlter) {
215         double amountToSubFromSide = mc.calculateAmountForCalories((toAlter / 4 * 3), meal.getSideIngredient());
216         double amountToSubFromMain = mc.calculateAmountForCalories(toAlter / 4, meal.getMainIngredient());
217         meal.setSideIngredientAmount(meal.getSideIngredientAmount() + amountToSubFromSide);
218         meal.setMainIngredientAmount(meal.getMainIngredientAmount() + amountToSubFromMain);
219     }
220 
221     private void alterCaloriesFromMain(double toAlter) {
222         double amountToSubFromMain = mc.calculateAmountForCalories(toAlter, meal.getMainIngredient());
223         meal.setMainIngredientAmount(meal.getMainIngredientAmount() + amountToSubFromMain);
224     }
