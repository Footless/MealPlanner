37      @Override
38      public void run() {
39          frame = new JFrame("Meal Planner");
40          frame.setPreferredSize(new Dimension(650, 500));
41          cardL = new CardLayout();

45          frame.setDefaultCloseOperation(WindowConstants.EXIT_ON_CLOSE);
46  
47          createComponents(frame.getContentPane());
48  
49          frame.setLocationByPlatform(true);
50          frame.pack();
51          frame.setVisible(true);
52      }

54      private void createComponents(Container contentPane) {
55          JPanel cards = new JPanel(cardL);
56          JPanel startCard = createStartCard(cards);
57          JPanel mainIngredientCard = createMainIngredientCard(cards);
58  //        JPanel mealCard = createMealCard(cards);
59          cards.add(startCard, "start");
60          cards.add(mainIngredientCard, "askMainIngredient");
61  //        cards.add(CealCard, "readyMeal");
62          contentPane.add(cards);
63      }

65      public JFrame getFrame() {
66          return frame;
67      }

69      private JPanel createStartCard(Container container) {
70          JPanel startCard = new JPanel();
71          startCard.setBackground(Color.black);
72          startCard.setLayout(new GridLayout(3, 1, 2, 2));
73          
74          JButton getMeal = createButton("Laske ateria");
75          StartNewMealListener snml = new StartNewMealListener(cardL, container);
76          getMeal.addActionListener(snml);
77          
78          JButton addIngredients = createButton("Lisää uusia raaka-aineita");
79          
80          JButton browseIngredients = createButton("Selaa raaka-aineita");
81          
82          startCard.add(getMeal);
83          startCard.add(addIngredients);
84          startCard.add(browseIngredients);
85          return startCard;
86      }

88      private JPanel createMainIngredientCard(JPanel cards) {
89          JPanel card = new JPanel();
90          card.setBackground(Color.black);
91          GridLayout layout = new GridLayout(4, 1);  // perusjako kortille
92          card.setLayout(layout);
93          JPanel selection = new JPanel(new GridLayout(3,1)); // ensimmäinen kolmasosa kolmeen osaan
94          
95          JTextArea instructions = createTextArea("Valitse listasta haluamasi raaka-aine");
96          JComboBox box = createProteinComboBox();                // lisätään ensimmäisen osan komponentit
97          JTextArea proteinInstructions = createTextArea("Valitse halaumasi määrä proteiinia:");
98          selection.add(instructions);
99          selection.add(box);
100         selection.add(proteinInstructions);
101         
102         JPanel restMacros = new JPanel(new GridLayout(3,1));
103         JSlider proteinSlider = createSlider(10, 60, 20);
104         JTextArea fatInstructions = createTextArea("Valitse haluamasi rasvan määrä");
105         JSlider fatSlider = createSlider(10, 50, 20);
106         restMacros.add(proteinSlider);
107         restMacros.add(fatInstructions);
108         restMacros.add(fatSlider);
109         
110         JPanel calories = new JPanel(new GridLayout(2, 1));
111         JTextArea caloriesInstructions = createTextArea("Valitse haluamasi määrä proteiinia:");
112         JSlider calorieSlider = createSlider(200, 2000, 400);
113         calories.add(caloriesInstructions);
114         calories.add(calorieSlider);
115         
116         JPanel buttons = new JPanel(new GridLayout(2, 1));
117         JButton ok = new JButton("Laske ateria");
118         SelectMainIngListener sml = new SelectMainIngListener(this, cardL, cards, box, proteinSlider, fatSlider, calorieSlider, mealCalculator);
119         ok.addActionListener(sml);
120         JButton back = new JButton("Takaisin alkuun");
121         BackButtonListener bbl = new BackButtonListener(cardL, cards);
122         back.addActionListener(bbl);
123         buttons.add(ok);
124         buttons.add(back);
125         card.add(selection);
126         card.add(restMacros);
127         card.add(calories);
128         card.add(buttons);
129         return card;
130     }
131     
132     protected JPanel createMealCard(Container container) {
133         JPanel card = new JPanel(new GridLayout(1, 1));
134         JTextArea meal = createTextArea(mealCalculator.getMeal().toString());
135         card.add(meal);
136         container.add(card);
137         container.add(card, "readyMeal");
138         return card;
139     }
140 
141     private JComboBox createProteinComboBox() {
142         List<String> mainsAsList = new ArrayList<>();
143         for (Ingredient ing : mealCalculator.getMainIngredients()) {
144             mainsAsList.add(ing.getName());
145         }
146         String[] mains = new String[mainsAsList.size()];
147         mains = mainsAsList.toArray(mains);
148         JComboBox box = new JComboBox(mains);
149         box.setFont(new Font("Arial", 1, 17));
150         box.setBackground(mainColor);
151         return box;
152     }
153 
154     private JTextArea createTextArea(String text) {
155         JTextArea instructions = new JTextArea(text);
156         Font f = new Font("Arial", 1, 18);
157         instructions.setFont(f);
158         return instructions;
159     }
160 
161     private JButton createButton(String text) {
162         JButton button = new JButton(text);
163         button.setBackground(Color.cyan);
164         return button;
165     }
166 
167     private JSlider createSlider(int min, int max, int set) {
168         JSlider slider = new JSlider(JSlider.HORIZONTAL, min, max, set);
169         slider.setMinorTickSpacing(min / 5);
170         slider.setMajorTickSpacing(min);
171         slider.setPaintTicks(true);
172         slider.setPaintLabels(true);
173         slider.setLabelTable(slider.createStandardLabels(min));
174         slider.setBackground(mainColor);
175         return slider;
176     }
