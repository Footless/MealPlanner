72      private void createComponents(Container contentPane) throws IOException {
73          JPanel cards = new JPanel(cardL);
74          JPanel startCard = createStartCard(cards);
75          cards.add(startCard, "start");
76          createMealCards(cards);
77  
78          contentPane.add(cards);
79      }

81      private JPanel createStartCard(Container container) {
82          JPanel startCard = new JPanel();
83          startCard.setBackground(Color.black);
84          startCard.setLayout(new GridLayout(3, 1, 2, 2));
85  
86          JButton getMeal = compFactory.createButton("Laske ateria");
87          SelectCardListener snml = new SelectCardListener(this, cardL, container, "askMainIngredient");
88          getMeal.addActionListener(snml);
89  
90          JButton addIngredients = compFactory.createButton("Lisää uusia raaka-aineita");
91  
92          JButton browseIngredients = compFactory.createButton("Selaa raaka-aineita");
93  
94          startCard.add(getMeal);
95          startCard.add(addIngredients);
96          startCard.add(browseIngredients);
97          return startCard;
98      }

100     private void createMealCards(JPanel cards) throws IOException {

103     }

109     private JPanel createMainIngredientCard(JPanel cards) throws IOException {
110         mealCalculator = new CalculateMeal();
111         helper = new MealCalcHelper(mealCalculator);
112         
113         JPanel card = new JPanel(new BorderLayout());
114         card.setBackground(Color.black);
115         JTextArea instructions = compFactory.createTextArea("Valitse listasta haluamasi raaka-aine");
116         card.add(instructions, BorderLayout.NORTH);
117         ButtonGroup mainsButtonGroup = compFactory.createMainsButtons(card, mealCalculator);
118         ActionListener sml = new SelectMainIngListener(this, cardL, mainsButtonGroup, cards, "askCalories");
119         addNextAndBackButtons(cards, card, "start", sml);
120         return card;
121     }

123     public JPanel createCaloriesCard(Container container) throws IOException {
124         JPanel card = new JPanel(new BorderLayout());
125         card.setBackground(Color.black);
126         JTextArea caloriesInstructions = compFactory.createTextArea("Valitse haluamasi määrä kaloreita:");
127         Ingredient ing = mealCalculator.getIngredients().get("mains").get(helper.getMainIngredientId());
128         double mul = getMultiplier(ing);
129         int min = (int) Math.ceil(ing.getCalories() * mul);
130         JSlider calorieSlider = compFactory.createSlider(min, 800, 800 / 2);
131         card.add(caloriesInstructions, BorderLayout.NORTH);
132         card.add(calorieSlider, BorderLayout.CENTER);
133         ActionListener scl = new SelectCaloriesListener(this, calorieSlider, container, "askProtein", cardL);
134         addNextAndBackButtons(container, card, "askMainIngredient", scl);
135         return card;
136     }

138     public JPanel createProteinsCard(Container container) {
139         JPanel card = new JPanel(new BorderLayout());
140         card.setBackground(Color.black);
141         JTextArea proteinInstructions = compFactory.createTextArea("Valitse haluamasi proteiinin määrä:");
142         Ingredient ing = mealCalculator.getIngredients().get("mains").get(helper.getMainIngredientId());
143         double mul = getMultiplier(ing);
144         int min = (int) Math.ceil(ing.getProtein() * mul);
145         JSlider proteinSlider = compFactory.createSlider(min, min * 5, min * 5 / 2);
146         card.add(proteinInstructions, BorderLayout.NORTH);
147         card.add(proteinSlider, BorderLayout.CENTER);
148         ActionListener spl = new SelectProtListener(this, proteinSlider, container, "askFat", cardL);
149         addNextAndBackButtons(container, card, "askCalories", spl);
150         return card;
151     }

153     public JPanel createFatsCard(Container container) {
154         JPanel card = new JPanel(new BorderLayout());
155         card.setBackground(Color.black);
156         JTextArea fatInstructions = compFactory.createTextArea("Valitse haluamasi rasvan määrä:");
157         Ingredient ing = mealCalculator.getIngredients().get("mains").get(helper.getMainIngredientId());
158         double mul = getMultiplier(ing);
159         int min = (int) Math.ceil(ing.getFat() * mul);
160         JSlider fatSlider = compFactory.createSlider(min, 40, 40 / 2);
161         card.add(fatInstructions, BorderLayout.NORTH);
162         card.add(fatSlider, BorderLayout.CENTER);
163         ActionListener sfl = new SelectFatListener(mealCalculator, this, fatSlider, container, "readyMeal", cardL);
164         addNextAndBackButtons(container, card, "askProtein", sfl);
165         return card;
166     }

168     public JPanel createReadyMealCard(Container container) {
169         JPanel card = new JPanel(new BorderLayout());
170         card.setBackground(Color.black);
171         JTextArea title = compFactory.createTextArea("Tässä ateriasi:");
172         compFactory.createMealText(card, mealCalculator, helper);
173         JPanel buttons = new JPanel(new GridLayout(2, 1));
174         JButton backToStart = compFactory.createButton("Valmis");
175         backToStart.setActionCommand("clear");
176         JButton back = compFactory.createButton("Takaisin");
177         ActionListener start = new SelectCardListener(this, cardL, container, "start");
178         ActionListener prevCardL = new SelectCardListener(this, cardL, container, "askFat");
179         back.addActionListener(prevCardL);
180         backToStart.addActionListener(start);
181         buttons.add(backToStart);
182         buttons.add(back);
183         card.add(buttons, BorderLayout.SOUTH);
184         card.add(title, BorderLayout.NORTH);
185         return card;
186     }
187 
188     private void addNextAndBackButtons(Container container, JPanel card, String prevCard, ActionListener al) {
189         JPanel buttons = new JPanel(new GridLayout(2, 1));
190         JButton next = compFactory.createButton("Seuraava");
191         JButton back = compFactory.createButton("Takaisin");
192         SelectCardListener prevCardL = new SelectCardListener(this, cardL, container, prevCard);
193         back.addActionListener(prevCardL);
194         ActionListener nextCardL = al;
195         next.addActionListener(nextCardL);
196         buttons.add(next);
197         buttons.add(back);
198         card.add(buttons, BorderLayout.SOUTH);
199     }
200 
201     private double getMultiplier(Ingredient ing) {
202         if (ing.getProtein() > 23) {
203             return 0.5;
204         } else {
205             return 1;
206         }
207     }
