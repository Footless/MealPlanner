114     private JPanel createMainIngredientCard(JPanel cards) throws IOException {
115         mealCalculator = new CalculateMeal();
116         helper = new MealCalcHelper();
117         JPanel card = new JPanel(new BorderLayout());
118         card.setBackground(Color.black);
119         JTextArea instructions = createTextArea("Valitse listasta haluamasi raaka-aine");
120         card.add(instructions, BorderLayout.NORTH);
121         ButtonGroup mainsButtonGroup = createMainsButtons(card);
122         ActionListener sml = new SelectMainIngListener(this, cardL, helper, mainsButtonGroup, cards, "askCalories");
123         addNextAndBackButtons(cards, card, "start", sml);
124         return card;
125     }

127     public JPanel createCaloriesCard(Container container) throws IOException {
128         JPanel card = new JPanel(new BorderLayout());
129         card.setBackground(Color.black);
130         JTextArea caloriesInstructions = createTextArea("Valitse haluamasi määrä kaloreita:");
131         Ingredient ing = mealCalculator.getIngredients().get("mains").get(helper.getMainIngredientId());
132         double mul = getMultiplier(ing);
133         int min = (int) Math.ceil(ing.getCalories() * mul);
134         JSlider calorieSlider = createSlider(min, 800, 800 / 2);
135         card.add(caloriesInstructions, BorderLayout.NORTH);
136         card.add(calorieSlider, BorderLayout.CENTER);
137         ActionListener scl = new SelectCaloriesListener(this, helper, calorieSlider, container, "askProtein", cardL);
138         addNextAndBackButtons(container, card, "askMainIngredient", scl);
139         return card;
140     }

172     public JPanel createReadyMealCard(Container container) {
173         JPanel card = new JPanel(new BorderLayout());
174         card.setBackground(Color.black);

207     private JSlider createSlider(int min, int max, int set) {
208         JSlider slider = new JSlider(JSlider.HORIZONTAL, min, max, set);
209         slider.setMinorTickSpacing(min);
210         slider.setMajorTickSpacing(max / 4);
211         slider.setPaintTicks(true);
212         slider.setPaintLabels(true);
213         slider.setLabelTable(slider.createStandardLabels(max / 10));
214         slider.setBackground(mainColor);
215         return slider;
216     }

218     private void addNextAndBackButtons(Container container, JPanel card, String prevCard, ActionListener al) {
219         JPanel buttons = new JPanel(new GridLayout(2, 1));
220         JButton next = createButton("Seuraava");
221         JButton back = createButton("Takaisin");
222         SelectCardListener prevCardL = new SelectCardListener(this, cardL, container, prevCard);
223         back.addActionListener(prevCardL);
224         ActionListener nextCardL = al;
225         next.addActionListener(nextCardL);
226         buttons.add(next);
227         buttons.add(back);
228         card.add(buttons, BorderLayout.SOUTH);
229     }

231     private ButtonGroup createMainsButtons(JPanel card) throws IOException {
232         List<Ingredient> mains = mealCalculator.getMainIngredients();
233         JPanel mainButtons = new JPanel(new GridLayout(mains.size(), 1));
234         mainButtons.setAutoscrolls(true);
235         ButtonGroup buttons = new ButtonGroup();
236         for (Ingredient main : mains) {
237             JRadioButton button = new JRadioButton(main.getName());
238             button.setBackground(mainColor);
239             button.setActionCommand(main.getName());
240             button.setSelected(true);
241             buttons.add(button);
242             mainButtons.add(button);
243             
244         }
245         mainButtons.setBackground(mainColor);
246         card.add(mainButtons, BorderLayout.CENTER);
247         return buttons;
248     }

250     private double getMultiplier(Ingredient ing) {
251         if (ing.getProtein() > 23) {
252             return 0.5;
253         } else {
254             return 1;
255         }
256     }

258     private void createMealText(JPanel card) {
259         Meal meal = mealCalculator.getMeal();
260         JPanel pane = new JPanel();
261         pane.setBackground(Color.cyan);
262         BoxLayout boxL = new BoxLayout(pane, BoxLayout.PAGE_AXIS);
263         pane.setLayout(boxL);
264         pane.add(Box.createRigidArea(new Dimension(0,10)));
265         JTextArea title = createTextArea("Komponentit ja määrät:");
266         pane.add(title);
267         pane.add(Box.createRigidArea(new Dimension(0,5)));
268         JTextArea main = createIngText(meal.getMainIngredient(), meal.getMainIngredientAmount());
269         main.setSize(400, 10);
270         pane.add(main);
271         JTextArea side = createIngText(meal.getSideIngredient(), meal.getSideIngredientAmount());
272         side.setSize(400, 10);
273         pane.add(side);
274         JTextArea sauce = createIngText(meal.getSauce(), meal.getSauceAmount());
275         pane.add(sauce);
276         JTextArea misc = createIngText(meal.getMisc(), meal.getMiscAmount());
277         pane.add(misc);
278         pane.add(Box.createRigidArea(new Dimension(0,5)));
279         JTextArea subTitle = createTextArea("Ravintosisältö");
280         pane.add(subTitle);
281         JTextArea calories = createMacroText("Kalorit", meal.getCalories(), helper.getDesiredCalories(), "kcal");
282         pane.add(calories);
283         JTextArea protein = createMacroText("Proteiini", meal.getProtein(), helper.getDesiredProtein(), "gr");
284         pane.add(protein);
285         JTextArea fat = createMacroText("Rasva", meal.getFat(), helper.getDesiredFat(), "gr");
286         pane.add(fat);
287         JTextArea carb = createMacroText("Hiilihydraatit", meal.getCarbs(), "gr");
288         pane.add(carb);
289         JTextArea fiber = createMacroText("Kuidut", meal.getFiber(), "gr");
290         pane.add(fiber);
291         card.add(pane, BorderLayout.CENTER);
292     }

298     private JTextArea createIngText(Ingredient ing, double amount) {
299         JTextArea jta = createTextArea(ing + ": " + Math.ceil(amount * 100) + "gr");
300         return jta;
301     }

303     private JTextArea createMacroText(String macroName, double macros, int desiredMacros, String unit) {
304         JTextArea jta = createTextArea(macroName + ": " + Math.ceil(macros) + unit + "(" + desiredMacros + unit + ")");
305         return jta;
306     }

308     private JTextArea createMacroText(String macroName, double macros, String unit) {
309         JTextArea jta = createTextArea(macroName + ": " + Math.ceil(macros) + unit);
310         return jta;
