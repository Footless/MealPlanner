111     private void createAllMacros(CalculateMeal cm, JPanel pane) {
112         Meal meal = cm.getMeal();
113         JLabel subTitle = createLabel("Ravintosisältö (suluissa halutut makrot)");
114         subTitle.setBackground(mainColor);
115         pane.add(subTitle);
116         JTextArea calories = createMacroText("Kalorit", meal.getCalories(), helper.getDesiredCalories(), "kcal");
117         pane.add(calories);
118         JTextArea protein = createMacroText("Proteiini", meal.getProtein(), helper.getDesiredProtein(), "gr");
119         pane.add(protein);
120         JTextArea fat = createMacroText("Rasva", meal.getFat(), helper.getDesiredFat(), "gr");
121         pane.add(fat);
122         JTextArea carb = createMacroText("Hiilihydraatit", meal.getCarbs(), "gr");
123         pane.add(carb);
124         JTextArea fiber = createMacroText("Kuidut", meal.getFiber(), "gr");
125         pane.add(fiber);
126     }

128     private void createAllIngs(CalculateMeal cm, JPanel pane) {
129         Meal meal = cm.getMeal();
130         JLabel title = createLabel("Komponentit ja määrät:");
131         pane.add(title);
132         pane.add(Box.createRigidArea(new Dimension(0, 5)));
133         JTextArea main = createIngText(meal.getMainIngredient(), meal.getMainIngredientAmount());
134         main.setSize(400, 10);
135         pane.add(main);
136         JTextArea side = createIngText(meal.getSideIngredient(), meal.getSideIngredientAmount());
137         side.setSize(400, 10);
138         pane.add(side);
139         JTextArea sauce = createIngText(meal.getSauce(), meal.getSauceAmount());
140         pane.add(sauce);
141         JTextArea misc = createIngText(meal.getMisc(), meal.getMiscAmount());
142         pane.add(misc);
143     }

145     protected void createMealText(JPanel card, CalculateMeal cm, MealCalcHelper helper) {
146         JPanel pane = new JPanel();

174     protected JSlider createSlider(int min, int max, int set) {
175         JSlider slider = new JSlider(JSlider.HORIZONTAL, min, max, set);
176         slider.setMinorTickSpacing(min);
177         slider.setMajorTickSpacing(max / 4);
178         slider.setPaintTicks(true);
179         slider.setPaintLabels(true);
180         slider.setLabelTable(slider.createStandardLabels(max / 10));
181         slider.setBackground(mainColor);
182         return slider;
183     }

185     protected ButtonGroup createIngButtons(JPanel card, CalculateMeal mealCalculator, String select) throws IOException {
186         List<Ingredient> ings;
187         if (select.contains("main")) {
188             ings = helper.getMainIngredients();
189         } else if (select.contains("side")) {
190             ings = helper.getSideIngredients();
191         } else {
192             ings = new ArrayList<>();
193         }
194         JPanel ingButtons = new JPanel(new GridLayout(ings.size() + 1, 1));
195         ingButtons.setAutoscrolls(true);
196         ButtonGroup buttons = new ButtonGroup();
197         if (select.contains("side")) {
198             JRadioButton randomButton = new JRadioButton("Satunnainen lisäke");
199             randomButton.setActionCommand("misc");
200             randomButton.setSelected(true);
201             buttons.add(randomButton);
202             ingButtons.add(randomButton);
203         }
204         for (Ingredient ing : ings) {
205             JRadioButton button = new JRadioButton(ing.getName());
206             button.setBackground(mainColor);
207             button.setName(ing.getName());
208             button.setActionCommand(ing.getName());
209             addToolTipForButton(button, select);
210             if (select.contains("main")) {
211                 button.setSelected(true);
212             }
213             buttons.add(button);
214             ingButtons.add(button);
215         }
216         ingButtons.setBackground(mainColor);
217         card.add(ingButtons, BorderLayout.CENTER);
218         return buttons;
219     }

221     protected void addNextAndBackButtons(Container container, JPanel card, String prevCard, ActionListener al) {
222         JPanel buttons = new JPanel(new GridLayout(2, 1));
223         JButton next = createButton("Seuraava");
224         JButton back = createButton("Takaisin");
225         SelectCardListener prevCardL = new SelectCardListener(cardL, container, prevCard);

232     }

241     /**
242      * Käytetään luomaan JList annetuista raaka-aineista. Sisältää myös
243      * ListSelecionListenerin luonnin.
244      *
245      * @param ings raaka-aineet tauluna
246      * @param searchFieldComp JPanel jonne valmis JList tökätään
247      * @return
248      * @throws IOException
249      * @see enableIngButtons
250      */
251     public JList createSearchIngList(String[] ings, JPanel searchFieldComp) throws IOException {
252         JList list = new JList(ings);
253         list.setSelectionMode(ListSelectionModel.SINGLE_SELECTION);
254         list.setLayoutOrientation(JList.VERTICAL);
255         list.setVisibleRowCount(-1);
256         Font f2 = new Font("Arial", 1, 15);
257         list.setFont(f2);
258         list.setCursor(new Cursor(0));
259         list.setBackground(mainColor);
260         list.addListSelectionListener((ListSelectionEvent e) -> {
261             if (e.getValueIsAdjusting() == false) {
262                 try {
263                     if (list.getSelectedIndex() == -1) {
264                         enableIngButtons(searchFieldComp, false);
265                     } else {
266                         enableIngButtons(searchFieldComp, true);
267                     }
268                 } catch (IOException ex) {
269                     Logger.getLogger(ComponentFactory.class.getName()).log(Level.SEVERE, null, ex);
270                 }
271             }
272         });
273         return list;
274     }

283     public void createActionListenersForButtons(JPanel searchFieldComp) throws IOException {
284         JPanel panel = (JPanel) searchFieldComp.getComponent(1);
285         Component[] buttons = panel.getComponents();
286         ActionListener btnListener = new AddToIngsListener(searchFieldComp);
287         for (Component button : buttons) {
288             JButton jbutton = (JButton) button;
289             jbutton.addActionListener(btnListener);
290         }
291     }

293     protected void createAddIngButtons(JPanel searchFieldComp) {
294         JPanel panel = new JPanel(new GridLayout(1, 3));
295         JButton addToMains = createButton("Lisää pääraaka-aineisiin");
296         addToMains.setActionCommand("main");
297         JButton addToSides = createButton("Lisää lisäkkeisiin");
298         addToSides.setActionCommand("side");
299         JButton addToSauces = createButton("Lisää kastikkeisiin");
300         addToMains.setEnabled(false);
301         addToSides.setEnabled(false);
302         addToSauces.setEnabled(false);
303         panel.add(addToMains);
304         panel.add(addToSides);
305         panel.add(addToSauces);
306         searchFieldComp.add(panel, BorderLayout.SOUTH);
307         panel.validate();
308         panel.repaint();
309     }

311     protected void enableIngButtons(JPanel searchFieldComp, boolean value) throws IOException {
312         JPanel panel = (JPanel) searchFieldComp.getComponent(1);
313         Component[] buttons = panel.getComponents();
314         for (Component button1 : buttons) {
315             JButton button = (JButton) button1;
316             button.setEnabled(value);
317         }
318     }

320     private void addToolTipForButton(JComponent component, String select) throws IOException {
321         Ingredient ing;
322         if (select.contains("main")) {
323             ing = helper.getMainIngredientsAsMap().get(helper.getIdForMainIng(component.getName()));
324         } else if (select.contains("side")) {
325             ing = helper.getSideIngredientsAsMap().get(helper.getIdForSideIng(component.getName()));
326         } else {
327             ing = null;
328         }
329 
330         searchHelper.addMacros(ing);
331         component.setToolTipText("Ravintoarvot per 100g: \n"
332                 + "Kalorit: " + Math.ceil(ing.getCalories()) + "kcal \n"
333                 + "Proteiini: " + Math.ceil(ing.getProtein()) + "g \n"
334                 + "Rasva: " + Math.ceil(ing.getFat()) + "g \n"
335                 + "Hiilihydraatit: " + Math.ceil(ing.getCarb()) + "g \n"
336                 + "Kuidut: " + Math.ceil(ing.getFiber()) + "g");
337     }
