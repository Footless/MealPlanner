43  /**
44   * Apuluokka GUI:lle, jossa metodit joilla luodaan erilaisia komponentteja, kuten
45   * buttoneja, tekstikenttiä ja slidereita.
46   * @author kari

48  public class ComponentFactory {
49  
50      public final Color mainColor = Color.white;
51  
52      public JTextArea createIngText(Ingredient ing, double amount) {
53          JTextArea jta = createTextArea(ing + ": " + Math.ceil(amount * 100) + "gr");
54          return jta;
55      }
56  
57      public JTextArea createMacroText(String macroName, double macros, int desiredMacros, String unit) {
58          JTextArea jta = createTextArea(macroName + ": " + Math.ceil(macros) + unit + " (" + desiredMacros + unit + ")");
59          return jta;
60      }
61  
62      public JTextArea createMacroText(String macroName, double macros, String unit) {
63          JTextArea jta = createTextArea(macroName + ": " + Math.ceil(macros) + unit);
64          return jta;
65      }
66  
67      public void createMealText(JPanel card, CalculateMeal cm, MealCalcHelper helper) {
68          Meal meal = cm.getMeal();
69          JPanel pane = new JPanel();
70          pane.setBackground(Color.cyan);
71          BoxLayout boxL = new BoxLayout(pane, BoxLayout.PAGE_AXIS);
72          pane.setLayout(boxL);
73          pane.add(Box.createRigidArea(new Dimension(0, 10)));
74          JTextArea title = createTextArea("Komponentit ja määrät:");
75          pane.add(title);
76          pane.add(Box.createRigidArea(new Dimension(0, 5)));
77          JTextArea main = createIngText(meal.getMainIngredient(), meal.getMainIngredientAmount());
78          main.setSize(400, 10);
79          pane.add(main);
80          JTextArea side = createIngText(meal.getSideIngredient(), meal.getSideIngredientAmount());
81          side.setSize(400, 10);
82          pane.add(side);
83          JTextArea sauce = createIngText(meal.getSauce(), meal.getSauceAmount());
84          pane.add(sauce);
85          JTextArea misc = createIngText(meal.getMisc(), meal.getMiscAmount());
86          pane.add(misc);
87          pane.add(Box.createRigidArea(new Dimension(0, 5)));
88          JTextArea subTitle = createTextArea("Ravintosisältö (suluissa halutut makrot)");
89          pane.add(subTitle);
90          JTextArea calories = createMacroText("Kalorit", meal.getCalories(), helper.getDesiredCalories(), "kcal");
91          pane.add(calories);
92          JTextArea protein = createMacroText("Proteiini", meal.getProtein(), helper.getDesiredProtein(), "gr");
93          pane.add(protein);
94          JTextArea fat = createMacroText("Rasva", meal.getFat(), helper.getDesiredFat(), "gr");
95          pane.add(fat);
96          JTextArea carb = createMacroText("Hiilihydraatit", meal.getCarbs(), "gr");
97          pane.add(carb);
98          JTextArea fiber = createMacroText("Kuidut", meal.getFiber(), "gr");
99          pane.add(fiber);
100         card.add(pane, BorderLayout.CENTER);
101     }

103     public JTextArea createTextArea(String text) {
104         JTextArea instructions = new JTextArea(text);
105         Font f = new Font("Arial", 1, 18);
106         instructions.setFont(f);
107         return instructions;
108     }
109 
110     public JButton createButton(String text) {
111         JButton button = new JButton(text);
112         button.setBackground(Color.cyan);
113         return button;
114     }
115 
116     public JSlider createSlider(int min, int max, int set) {
117         JSlider slider = new JSlider(JSlider.HORIZONTAL, min, max, set);
118         slider.setMinorTickSpacing(min);
119         slider.setMajorTickSpacing(max / 4);
120         slider.setPaintTicks(true);
121         slider.setPaintLabels(true);
122         slider.setLabelTable(slider.createStandardLabels(max / 10));
123         slider.setBackground(mainColor);
124         return slider;
125     }
126 
127     public ButtonGroup createMainsButtons(JPanel card, CalculateMeal mealCalculator) throws IOException {
128         List<Ingredient> mains = mealCalculator.getMainIngredients();
129         JPanel mainButtons = new JPanel(new GridLayout(mains.size(), 1));
130         mainButtons.setAutoscrolls(true);
131         ButtonGroup buttons = new ButtonGroup();
132         for (Ingredient main : mains) {
133             JRadioButton button = new JRadioButton(main.getName());
134             button.setBackground(mainColor);
135             button.setActionCommand(main.getName());
136             button.setSelected(true);
137             buttons.add(button);
138             mainButtons.add(button);
139 
140         }
141         mainButtons.setBackground(mainColor);
142         card.add(mainButtons, BorderLayout.CENTER);
143         return buttons;
144     }
