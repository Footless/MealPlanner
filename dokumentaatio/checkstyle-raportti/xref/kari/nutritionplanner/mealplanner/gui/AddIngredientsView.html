19  import java.awt.BorderLayout;
20  import java.awt.CardLayout;
21  import java.awt.Color;
22  import java.awt.Cursor;
23  import java.awt.Font;
24  import java.awt.event.ActionListener;
25  import javax.swing.BoxLayout;
26  import javax.swing.DefaultListModel;
27  import javax.swing.JButton;
28  import javax.swing.JLabel;
29  import javax.swing.JList;
30  import javax.swing.JPanel;
31  import javax.swing.JScrollPane;
32  import javax.swing.JTextField;
33  import javax.swing.ListSelectionModel;
34  import javax.swing.event.ListSelectionEvent;
35  import kari.nutritionplanner.mealplanner.gui.controllers.SelectCardListener;
36  import kari.nutritionplanner.mealplanner.gui.controllers.SearchIngListener;
37  
38  /**
39   * Uusien ruoka-aineiden lisäämiselle oma luokkansa, osa graafista
40   * käyttöliittymää.
41   *
42   * @author kari
43   */
44  public class AddIngredientsView {
45  
46      private final CardLayout cardL;
47      private final ComponentFactory compFactory;
48      private final Color mainColor = Color.white;
49  
50      /**
51       * Konstuktori. Saa parametrinä käyttöliittymän käyttämän CardLayout-olion.
52       *
53       * @param cardL CardLayout-olio, jota käyttöliittymä käyttää näkymien
54       * vaihtamiseen
55       * @param compFactory Komponenttien tekoa varten ComponentFactory
56       */
57      public AddIngredientsView(CardLayout cardL, ComponentFactory compFactory) {
58          this.cardL = cardL;
59          this.compFactory = compFactory;
60      }
61  
62      protected JPanel createSearchIngCard(JPanel cards) {
63          JPanel card = new JPanel(new BorderLayout());
64          JLabel searchIngs = compFactory.createLabel("Hae raaka-aineita");
65          card.add(searchIngs, BorderLayout.NORTH);
66  
67          JPanel searchFieldComp = new JPanel(new BorderLayout());
68  
69          // hakutulokset esittävä kenttä. Samassa metodissa tehdään myös yläreunan hakukenttä ja nappula
70          createSearchResultsScrollPane(searchFieldComp);
71  
72          // alareunan napit, joilla raaka-aineita voi lisäillä tietokantaan
73          compFactory.createAddIngButtons(searchFieldComp);
74  
75          card.add(searchFieldComp, BorderLayout.CENTER);
76  
77          ActionListener al = new SelectCardListener(cardL, cards, "start");
78          compFactory.addNextAndBackButtons(cards, card, "start", al);
79  
80          return card;
81      }
82  
83      private void createFieldAndButton(JPanel searchFieldComp, DefaultListModel listModel) {
84          JPanel searchFieldAndButton = new JPanel();
85          BoxLayout boxL = new BoxLayout(searchFieldAndButton, BoxLayout.LINE_AXIS);
86          searchFieldAndButton.setLayout(boxL);
87          JTextField searchField = compFactory.createSearchField();
88  
89          JButton searchButton = compFactory.createButton("Hae");
90          ActionListener searchIngListener = new SearchIngListener(searchField, listModel, compFactory.getSearchHelper());
91          searchButton.addActionListener(searchIngListener);
92          searchField.addActionListener(searchIngListener);
93          searchFieldAndButton.add(searchField);
94          searchFieldAndButton.add(searchButton);
95          searchFieldComp.add(searchFieldAndButton, BorderLayout.NORTH);
96      }
97  
98      private void createSearchResultsScrollPane(JPanel searchFieldComp) {
99          DefaultListModel listModel = new DefaultListModel();
100         JList searchResults = new JList(listModel);
101         searchResults.setSelectionMode(ListSelectionModel.SINGLE_SELECTION);
102         searchResults.setLayoutOrientation(JList.VERTICAL);
103         searchResults.setVisibleRowCount(-1);
104         JScrollPane scrollResults = new JScrollPane(searchResults);
105         scrollResults.getVerticalScrollBar().setUnitIncrement(10);
106         scrollResults.getHorizontalScrollBar().setUnitIncrement(10);
107         Font f2 = new Font("Arial", 1, 15);
108         searchResults.setFont(f2);
109         searchResults.setCursor(new Cursor(0));
110         searchResults.setBackground(mainColor);
111         searchFieldComp.add(scrollResults, BorderLayout.CENTER);
112         searchResults.addListSelectionListener((ListSelectionEvent e) -> {
113             if (e.getValueIsAdjusting() == false) {
114                 if (searchResults.getSelectedIndex() == -1) {
115                     compFactory.enableIngButtons(searchFieldComp, false);
116                 } else {
117                     compFactory.enableIngButtons(searchFieldComp, true);
118                 }
119             }
120         });
121         createFieldAndButton(searchFieldComp, listModel);
122     }
123 }
