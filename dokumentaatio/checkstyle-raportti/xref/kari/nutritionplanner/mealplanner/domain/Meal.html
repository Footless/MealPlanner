17  public class Meal {
18  
19      private final Map<Integer, Ingredient> ingredients;
20      private final Map<Integer, Double> amounts;
21      
22      public Meal() {
23          this.ingredients = new HashMap<>();
24          this.amounts = new HashMap<>();
25          this.amounts.put(1, 0.0);
26          this.amounts.put(2, 0.0);
27          this.amounts.put(3, 0.0);
28          this.amounts.put(4, 0.0);
29          this.ingredients.put(1, new Ingredient(0, "dump"));
30          this.ingredients.put(2, new Ingredient(0, "dump"));
31          this.ingredients.put(3, new Ingredient(0, "dump"));
32          this.ingredients.put(4, new Ingredient(0, "dump"));
33      }
34  
35      public void setMainIngredient(Ingredient mainIngredient) {
36          ingredients.replace(1, mainIngredient);
37      }
38  
39      public void setSideIngredient(Ingredient sideIngredient) {
40          ingredients.replace(4, sideIngredient);
41      }
42  
43      public void setSauce(Ingredient sauce) {
44          ingredients.replace(2, sauce);
45      }
46  
47      public void setMisc(Ingredient misc) {
48          ingredients.replace(3, misc);
49      }
50  
51      public void setMainIngredientAmount(double mainIngredientAmount) {
52          this.amounts.replace(1, mainIngredientAmount);
53      }
54  
55      public void setSideIngredientAmount(double sideIngredientAmount) {
56          this.amounts.replace(4, sideIngredientAmount);
57      }
58  
59      public void setSauceAmount(double sauceAmount) {
60          this.amounts.replace(2, sauceAmount);
61      }
62  
63      public void setMiscAmount(double miscAmount) {
64          this.amounts.replace(3, miscAmount);
65      }
66  
67      public double getProtein() {
68          double protein = 0;
69          for (Integer i : ingredients.keySet()) {
70              protein += ingredients.get(i).getProtein() * amounts.get(i);
71          }
72          return protein;
73      }
74  
75      public double getFat() {
76          double fat = 0;
77          for (Integer i : ingredients.keySet()) {
78              fat += ingredients.get(i).getFat() * amounts.get(i);
79          }
80          return fat;
81      }
82  
83      public double getCalories() {
84          double calories = 0;
85          for (Integer i : ingredients.keySet()) {
86              calories += ingredients.get(i).getCalories() * amounts.get(i);
87          }
88          return calories;
89      }
90  
91      public double getCarbs() {
92          double carbs = 0;
93          for (Integer i : ingredients.keySet()) {
94              carbs += ingredients.get(i).getCarb() * amounts.get(i);
95          }
96          return carbs;
97      }
98  
99      public double getFiber() {
100         double fibers = 0;
101         for (Integer i : ingredients.keySet()) {
102             fibers += ingredients.get(i).getFiber() * amounts.get(i);
103         }
104         return fibers;
105     }
